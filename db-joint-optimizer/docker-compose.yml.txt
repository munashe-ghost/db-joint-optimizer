version: '3.8'

services:
  postgres:
    image: postgres:15
    container_name: db_postgres
    environment:
      POSTGRES_PASSWORD: postgres
      POSTGRES_DB: tpch
    ports:
      - "5432:5432"
    volumes:
      - ./postgres/init.sql:/docker-entrypoint-initdb.d/init.sql

  postgres_exporter:
    image: prometheuscommunity/postgres-exporter
    container_name: pg_exporter
    environment:
      DATA_SOURCE_NAME: "postgresql://postgres:postgres@postgres:5432/tpch?sslmode=disable"
    ports:
      - "9187:9187"
    depends_on:
      - postgres

  prometheus:
    image: prom/prometheus
    container_name: db_prometheus
    ports:
      - "9090:9090"
    volumes:
      - ./prometheus.yml:/etc/prometheus/prometheus.yml

  grafana:
    image: grafana/grafana
    container_name: db_grafana
    ports:
      - "3000:3000"
